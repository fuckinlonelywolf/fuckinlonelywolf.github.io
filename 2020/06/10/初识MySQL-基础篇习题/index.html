This is a basic exercise of learning MySQL

查询没有奖金，且工资小于18000的salary, last_name

SELECT
    last_name,
    salary 
FROM
    employees 
WHERE
    commission_pct IS NULL 
    AND salary < 18000;

查询employees表中，job_id不为’IT’或者工资为12000的员工信息

SELECT
    * 
FROM
    employees 
WHERE
    job_id <> 'IT' 
    OR salary = 12000;

查看departments表的结构

desc departments

查询部门departments表中涉及到了哪些位置编号

结合上一题，有个字段是location_id，查它。

SELECT DISTINCT
    location_id 
FROM
    departments;

查询员工的姓名和部门号的年薪，按年薪降序，按姓名升序

SELECT
    last_name,
    department_id,
    salary * 12 * (1+IFNULL( commission_pct, 0 )) 年薪 
FROM
    employees 
ORDER BY
    年薪 DESC,
    last_name ASC

选择工资不在8000和17000的员工的姓名和工资，按工资降序

SELECT
    last_name,
    salary

CASE

查询各部门中工资比本部门平均工资高的员工号，姓名，工资

这题有点难，我的话不分步反应不过来

#1.各部门的平均工资：
SELECT
    department_id,
    AVG( salary ) 
FROM
    employees 
GROUP BY
    department_id;
#2.加一个连接查，用department_id连接两个表：
SELECT
    employee_id,
    last_name,
    salary,
    e.department_id 
FROM
    employees e
    LEFT JOIN ( SELECT department_id, AVG( salary ) ag_sal FROM employees GROUP BY department_id ) d ON e.department_id = d.department_id 
WHERE
    e.salary > d.ag_sal;

总结一般的查询语句

select 查询列表
from 表
连接类型 join 表2
on 连接条件
where 筛选条件
group by 分组列表
having 分组后的筛选
order by 排序列表
limit 偏移、条目数
